### TIPI DI VARIABILE ###
int -> Numero Intero
real -> Numero Reale
txt -> Testo
bit -> Booleano

### OPERAZIONI ###
+ -> Somma
- -> Sottrazione
* -> Moltiplicazione
/ -> Divisione

++ -> sommare 1
-- -> sottrarre 1
is+ -> assegnare e sommare 1
is- -> assegnare e sottrarre 1

and -> AND
or -> OR
not -> NOT

gr -> >
le -> <
greq -> >=
leq -> <=
eq -> == (CONFRONTO)
is -> = (ASSEGNAZIONE)

; -> Terminare istruzione

do / done -> Iniziare e Terminare blocco
else -> Per continuare l'if
<> -> parentesi per condizioni
if -> blocco if
re -> while

in >> nome_variabile -> input da tastiera
out << "stringa" -> output a video


### ESEMPIO ###
real numeroReale;
txt stringa;

in >> numeroReale;

if < numeroReale greq 10 > do
	out >> “Numero maggiore di 10”
done

if < numeroReale eq 10 > do
	numeroReale is+ 10;
else
	numeroReale is 0;
done

re < numeroReale eq 10 > do
	numeroReale++;
done
